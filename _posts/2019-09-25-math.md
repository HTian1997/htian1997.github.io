### 7. Reverse Integer

```java
class Solution {
    public int reverse(int x) {
        int rev = 0;
        while(x != 0){
            int pop = x % 10;
            x /= 10;
            if(rev > Integer.MAX_VALUE / 10 || (rev == Integer.MAX_VALUE / 10 && pop > Integer.MAX_VALUE % 10)){
                return 0;
            }else if(rev < Integer.MIN_VALUE / 10 || (rev == Integer.MIN_VALUE / 10 && x < Integer.MIN_VALUE % 10)){
                return 0;
            }
            rev = rev * 10 + pop;
        }
        return rev;
    }
}
```

### 165. Compare Version Numbers

```java
class Solution {
    public int compareVersion(String version1, String version2) {
        String[] v1 = version1.split("\\.");
        String[] v2 = version2.split("\\.");
        
        int max_length = Math.max(v1.length, v2.length);
        int v1_num, v2_num;
        for(int i = 0; i < max_length; i++){
            v1_num = i < v1.length ? Integer.parseInt(v1[i]) : 0;
            v2_num = i < v2.length ? Integer.parseInt(v2[i]) : 0;
            if(v1_num == v2_num)
                continue;
            else if(v1_num > v2_num)
                return 1;
            else
                return -1;
        }
        
        return 0;
    }
}
```

### 66. Plus One

```java
class Solution {
    public int[] plusOne(int[] digits) {
        int carry = 1;
        //int index = digits.length() - 1;
        for(int i = digits.length-1; i >= 0; i--){
            digits[i] += carry;
            if (digits[i] >= 10){
                carry = 1;
                digits[i] -= 10;
            }
            else
                break;
        }
        if (digits[0] == 0){
            int[] ans = new int[digits.length + 1];
            ans[0] = 1;
            ans[1] = 0;
            for (int i = 2; i < ans.length; i++)
                ans[i] = digits[i-1];
            return ans;
        }
        else
            return digits;
    }
}
```

### 8. String to Integer

```java
class Solution {
    public int myAtoi(String str) {
        int start = 0, end = 0, length = str.length();
        boolean neg = false;
        for (start = 0; start < length; start++){
            if (str.charAt(start) >= '0' && str.charAt(start) <= '9'){
                break;
            }else if (str.charAt(start) == '-' || str.charAt(start)  == '+'){
                neg = (str.charAt(start) == '-');
                start ++;
                break;
            }else if (str.charAt(start) != ' '){
                return 0;
            }
        }
        
        for (end = start; end < length; end++){
            if (str.charAt(end) < '0' || str.charAt(end) > '9'){
                break;
            }
        }
        
        int ans = 0, cur = 0;
        String num = str.substring(start, end);
        for (int x = 0; x < num.length(); x ++){
            cur = num.charAt(x) - '0';
            if (neg){
                if (ans < Integer.MIN_VALUE / 10 || (ans == Integer.MIN_VALUE / 10 && cur > 8)){
                    return Integer.MIN_VALUE;
                }
                ans = ans * 10 - cur;
            }else{
                if (ans > Integer.MAX_VALUE / 10 || (ans == Integer.MAX_VALUE / 10 && cur > 7)){
                    return Integer.MAX_VALUE;
                }
                ans = ans * 10 + cur;
            }
        }
        
        return ans;
    }
}
```

### 258. Add Digits

```java
class Solution {
    public int addDigits(int num) {
        if(num == 0)
            return 0;
        int ans = num % 9;
        if(ans != 0)
            return ans;
        else
            return 9;
    }
```

### 69. Sqrt(x)

```java
class Solution {
    public int mySqrt(int x) {
        long a = x;
        while (a * a > x) {
            a = (a + x / a) / 2;
        }
        return (int)a;
    }
}
```

### 50. Pow(x,n)

```java
class Solution {
    public double myPow(double x, int n) {
        long N = n;
        if (N < 0){
            x = 1 / x;
            N = -N;
        }
        return pow(x, N);
    }
    
    private double pow(double x, long n){
        if(n == 0)
            return 1.0;
        double half = pow(x, n/2);
        if (n % 2 == 0)
            return half * half;
        else
            return half * half * x;
    }
}
```

### 367. Valid Perfect Square

```java
class Solution {
    public boolean isPerfectSquare(int num) {
        if (num == 1)
            return true;
        int i = num / 2;
        while((double)i * i > num){
            i = (i + num / i) / 2;
        }
        return i*i == num;
    }
}
```

### 365. Water and Jug Problem

```java
class Solution {
    public boolean canMeasureWater(int x, int y, int z) {
        return z == 0 || ( x + y >= z && z % gcd( x , y ) == 0);
    }
    
    private static int gcd(int number1, int number2) {
        if(number2 == 0){
            return number1;
        }
        return gcd(number2, number1%number2);
    }
}
```
